//显示整数的二进制
#include <stdio.h>
#include <stdlib.h>

void bitDis(int data)
{
    int i = 32;
    while(i--){
        if(data & (1<<i))
            printf("1");
        else
            printf("0");
        if(i%8 == 0)
            putchar(' ');
    }
    putchar(10);
}

int main()
{
    bitDis(91);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, i, j;
    int t;
    printf("请输入一个整数：n = ");
    scanf_s("%d", &n);
    t = 2 * n - 1;
    for (i = 1; i <= n; i++)                    //i控制行数
    {
        for (j = 1; j <= t - 2 * i; j++)                //填充空格
            putchar(0);
        for (j = 1; j <= i; j++)
        {
            if (i == n && j == 1)            //到了最后一行第一个
            {
                printf("%d", j);
            }
            else {
                printf(" %d", j);
            }
        }
        for (j = i - 1; j >= 1; j--)                //金字塔后半部分
            printf(" %d", j);
        putchar(10);
    }
    system("pause");
    return 0;
}

水仙花数

//水仙花数
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

unsigned int shuoming(unsigned int n)
{
    int i,
        t = n,
        len = 0,
        total = 0,
        p[10];

    while (t / 10)
    {
        t /= 10;
        len++;
    }

    len++;
    t = n;

    for (i = 0; i < len; i++)
    {
        p[i] = t % 10;
        t /= 10;
        p[i] = (unsigned int)pow(p[i], len);
        total += p[i];
    }

    if (total == n)
    {
        printf("the %d bits number is %u\n", len,n);
    }

    return n;
}

int main()
{
    unsigned int n;

    printf("pls input one integer:");
    scanf_s("%u", &n);

    for (n; n >10; n--)
    {
        shuoming(n);
    }

    system("pause");
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main() {
    int i,j,
        a_array[3][3] = {
        {0,1,2},
        {10,11,12},
        {20,21,22}
    };
    int* p = a_array;
    printf("0x%x; 0x%x\n", p, &a_array);
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++)
        {
            printf("%d", p[j]);

    system("pause");
    return 0;
}

数字获取

c语言编程一定要严谨，173行calloc前没有标注int*，是不会报错的，但是在Dev-C++中获得的结果是错误的。2020年6月2日22点53分

#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>

int get_num_from_char(char* str)
{
    int i,
        j = 0,
        re = 0;

    int* ptr = (int*)calloc(10, sizeof(int));
    if (NULL == ptr)
        return -1;

    for (i = 0; i < (int)strlen(str); i++)
    {
        if (str[i] >= '0' && str[i] <= '9') {
            ptr[j] = (int)(str[i] - '0');
            j++;
        }
    }

    j--;

    for (i = 0; i < sizeof(ptr); i++) {
        re += (int)(ptr[i] * pow(10, (j--)));
    }

    free(ptr);
    ptr = NULL;
    return re;
}

int main()
{
    int a = 0;
    char buff[1024] = {0};

    FILE* fp = fopen("gg.txt", "r");
    if(NULL == fp)
    {
        printf("文件打开失败！\n");
        return -1;
    }

    while (fgets(buff, 20, fp) != NULL)
    {
        a += get_num_from_char(buff);
    }

    printf("a = %d\n", a);
    fclose(fp);
    system("pause");
    return 0;
}
