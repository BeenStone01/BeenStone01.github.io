命令

常用命令

[been@linux7 ~]$ pwd                          //显示当前路径
[been@linux7 ~]$ ls /home/vagrant             //罗列目录
[been@linux7 ~]$ ls -l                        //显示详细信息（不包含.开头的文件）
[been@linux7 ~]$ ls -al /home/vagrant/code    //显示详细信息（包含.开头的文件）
[been@linux7 ~]$ chown users:vagrant /home/vagrant/.composer      //变更所有者即用户组
[been@linux7 ~]$ cd /home/vagrant/code               //跳转
[been@linux7 ~]$ mkdir /home/been/software/ntfs      //创建空目录
[been@linux7 ~]$ rmdir /home/been/usb                 //移除空目录
[been@linux7 ~]$ sudo mkdir -p /mnt/u/CT/down            //创建目录树
[been@linux7 ~]$ sudo mv /home/ntfs-3g/* /home/been/software/ntfs //移动ntfs-3g内的所有文件到ntfs内
[been@linux7 ~]$ cp /文档/Linux_Guide.docx /图片/Linux_guide.docx //复制并重命名文件
[been@linux7 ~]$ cp -r   static  ../我的工具                      //递归复制 

[been@linux7 ~]$ rm -f [文件]                                            //删除文件？强制删除？
[been@linux7 ~]$ rm -r [文件夹]                                          //删除文件夹
[been@linux7 ~]$ rm -rf /mnt/usb_disk                                    //删除目录

软件安装

[been@linux7 ~]# rpm -qa|grep samba             //查看已安装软件信息 
[been@linux7 ~]# rpm -qi samba                  //查看软件版本信息
//           /usr/src/ 软件源码一般放置在

[been@192 Downloads]$ xz -d glib-2.63.0.tar.xz
[been@192 Downloads]$ tar -xvf glib-2.63.0.tar      //.xz文件解压分两步

[been@linux7 ~]$ sudo tar zxvf /home/been/下载/ntfs-3g_ntfsprogs-2017.3.23.tgz -C ..  //当前目录的上一级
[been@linux7 ~]$ sudo tar -zxvf ***.tar.gz
[been@linux7 ~]$ sudo tar -jxvf ***.tar.bz(或bz2)
[been@linux7 ~]$ sudo ./configure --prefix=/usr/local/ntfs-3g //编译文件 指定安装位置
[been@linux7 ~]$ Sudo make
[been@linux7 ~]$ Sudo make install  //安装文件

关机命令

halt                        //立刻关机  (结束进程……关机，推荐使用)
poweroff                    //立刻关机
shutdown -h now             //立刻关机(root用户使用) 
shutdown -h 10              10秒后关机
shutdown +5 "System will shutdown after 5 minutes"
reboot                      //重启
logout                      //注销

rpm使用

rpm -ivh ***.rpm　　//其中i表示安装，v表示显示安装过程，h表示显示进度
rpm -Uvh ***.rpm      //update
//删除软件包
rpm -e PACKAGE_NAME
rpm -e –nodeps PACKAGE_NAME    #不考虑依赖包
rpm -e –allmatches PACKAGE_NAME    #删除所有跟PACKAGE_NAME匹配的所有版本的包
//查询软件包
# rpm -q PACKAGE_NAME
# rpm -qp ***.rpm 获取当前目录下的rpm包相关信息
# rpm -qa | less 列出所有已安装的软件包
# rpm -qf /usr/sbin/httpd 查看某个文件属于哪个软件包，可以是普通文件或可执行文件，跟文件的绝对路径
# rpm -qi PACKAGE_NAME 列出已安装的这个包的标准详细信息
# rpm -ql PACKAGE_NAME 列出rpm包的文件内容
# rpm -q –scripts kernel | less 列出已安装rpm包自带的安装前和安装后脚本
# rpm -qa –queryformat ‘Package %{NAME} was build on %{BUILDHOST}\n’ |less queryformat强大的查询
# rpm –querytags | less 可以列出queryformat可以使用的所有变量从而组合成更强大的查询
