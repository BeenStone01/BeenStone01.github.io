文件写入

文件写入（c：）

#include <stdio.h>
#include <string.h>
#include <stdlib.h>    

int main(int argc, char *argv[]) {

    FILE* fp = fopen("sugarball.txt","w+");
    if(NULL == fp)
        return -1;
    for(char ch = 'a';ch<='z';ch++)
    {
        printf("%c\n",fputc(ch,fp));         //每次写入一个字符 
    }

    printf("the position is %d, pr = %d\n",ftell(fp),feof(fp));        //26，0 

    rewind(fp);
    printf("rewind(fp), the position is %d, pr = %d\n",ftell(fp),feof(fp));        //0，0 

    char ret;

    while((ret = fgetc(fp)) && (!feof(fp)))            //每次读取一个字符 
    {
        printf("ret = %c\n",ret);
    }

    printf("the position is %d, pr = %d\n",ftell(fp),feof(fp));            //26，16 

    rewind(fp);
    printf("rewind(fp), the position is %d, pr = %d\n",ftell(fp),feof(fp));        //0，0 

    fseek(fp,0,SEEK_END);
    printf("fseek(fp,0,SEEK_END), the position is %d, pr = %d\n",ftell(fp),feof(fp));    //26，0 

    fclose(fp);
    fp = NULL;
    return 0;
}

    //文件写入

    FILE* fp = NULL;
    fp = fopen("./write.txt", "ab");    //二进制文件“rb" 只读，"wb"覆盖，”ab" 追加，r+,w+,a+
    if (fp == NULL) {
        printf("Failed!\n");
    }
    else {
        char* write_data = "this is a test!";
        fwrite(write_data, 1, strlen(write_data), fp);    //一次写入一个字符
        fprintf(fp, "10+3=%d", 13);     //或者格式化写入
        fputs("this is a test", fp);    //或者一次写入一行   fgets()   fscanf()详见库函数收录  
        fclose(fp);      //fflush(fp)
    }

}

#include <stdlib.h>
#include <direct.h>

{

    int position = ftell(f);                       //文件指针位置
    printf("The position is %d\n", position);
    fseek(f, 0, SEEK_END);                          //跳至文件末尾
    int file_size = ftell(f);                       //当前位置即为文件大小
    printf("The size of file is %d\n", file_size);
    char* file_data = malloc(file_size + 1);        //申请堆空间
    fseek(f, 0, SEEK_SET);                          //跳至文件开始
    int readed = fread(file_data, 1, file_size, f);     //每次读取一个字符，读取全部文字
    printf("read byte = %d\n",readed);
    file_data[file_size] = 0;                       //末尾置0
    printf("%s\n", file_data);
    fclose(f);
}

//注意要多申请一个空间位置！！
void str_Read(CStringArray& szArray)
{
    CFile file;
    if (file.Open(_T("./DeviceList.txt"), CFile::modeRead | CFile::shareDenyNone))
    {
        ULONGLONG len = file.GetLength();
        char* ptr = new char[len+1];
        memset(ptr, 0, len+1);
        file.Read(ptr, len);
        CString str(ptr);
        str_Split(str, szArray, _T("\r\n"));
        int nCount = szArray.GetSize();
        for (int i = 0; i < nCount; i++)
        {
            CString szUrl = szArray.GetAt(i);
            USES_CONVERSION;
            printf("%s\n", T2A(szUrl.GetString()));
        }
        file.Close();
        delete ptr;
    }
}

int CMFCDlg::InitListCtrl()
{
    int nCount = 0;
    CString str;
    CFile file;
    if (!file.Open(_T("./MediaData.txt"), CFile::modeRead | CFile::shareDenyNone))
    {
        AfxMessageBox(_T("播放列表初始化失败！"));
    }
    else
    {
        Media media;
        memset(&media, 0, sizeof(Media));
        while (sizeof(Media) == file.Read(&media,sizeof(Media)))
        {
            str.Format(_T("%d"), nCount + 1);
            m_list.InsertItem(nCount, str);
            Media* pMedia = new Media;
            memcpy(pMedia, &media, sizeof(Media));
            m_list.SetItemText(nCount, 1, CString(pMedia->sName));
            m_list.SetItemData(nCount, (DWORD_PTR)pMedia);
            nCount++;
        }
        file.Close();
    }
    return nCount;
}
