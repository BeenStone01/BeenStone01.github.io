SSDP开发说明

Been Stone 2020-04-23 17:08:18

开发步骤

一、获取设备信息

1. 方式一：主动搜寻

UDP方式向地址”239.255.255.250”的1900端口发送广播信息，查询设备信息。

char bufSend[] = "M-SEARCH * HTTP/1.1\r\n\
HOST: 239.255.255.250:1900\r\n\
MAN: \"ssdp:discover\"\r\n\
MX: 5\r\n\
ST: urn:schemas-upnp-org:service:AVTransport:1\r\n\
\r\n";

获取结果示例：

2. 方式二：被动侦听

加入组播
侦听本地1900端口

//端口复用
bool bReuseaddr = true;
ret = setsockopt(newSock, SOL_SOCKET, SO_REUSEADDR, (const char*)&bReuseaddr, sizeof(bReuseaddr));
if (0 != ret)
{
    fprintf(stderr, "%d setsockopt error1.\n", WSAGetLastError());
    return;
}

//加入组播
struct ip_mreq multiCast;
inet_pton(AF_INET, m_sHost.c_str(), &multiCast.imr_interface.S_un.S_addr);
inet_pton(AF_INET, "239.255.255.250", &multiCast.imr_multiaddr.S_un.S_addr);
if (!sock.SetOpt(IPPROTO_IP, IP_ADD_MEMBERSHIP, &multiCast, sizeof(multiCast)))
{
    cout << "Monitor().SetOpt2.Error:" << GetLastError() << endl;
    return -1;
}

//绑定本地端口
SOCKADDR_IN servAddr;
memset(&servAddr, 0, sizeof(servAddr));
servAddr.sin_family = AF_INET;
inet_pton(AF_INET, INADDR_ANY, &servAddr.sin_addr);
servAddr.sin_port = htons(1900);
ret = bind(newSock, (SOCKADDR*)&servAddr, sizeof(SOCKADDR));

二、获取设备描述xml文件

Http的Get方法访问设备获取设备详细信息描述的xml文件

三、解析xml获取设备信息

1. 解析设备描述，

<service>
<serviceType>urn:schemas-upnp-org:service:AVTransport:1</serviceType>
<serviceId>urn:upnp-org:serviceId:AVTransport</serviceId>
<SCPDURL>_urn-schemas-upnp-org-service-AVTransport_scpd.xml</SCPDURL>
<controlURL>_urn-schemas-upnp-org-service-AVTransport_control</controlURL>
<eventSubURL>_urn-schemas-upnp-org-service-AVTransport_event</eventSubURL>
</service>

获取AVTransport服务SCPDURL、controlURL、eventSubURL字段内的网址信息：

_urn-schemas-upnp-org-service-AVTransport_scpd.xml
_urn-schemas-upnp-org-service-AVTransport_control
_urn-schemas-upnp-org-service-AVTransport_event

BSTR DevicePlayer::GetXML(CString szUrl)
{
    BSTR bstrText = NULL;
    CString szContent;
    HRESULT hr;
    IXMLHttpRequestPtr request = NULL;

    if (FAILED(request.CreateInstance("MSXML2.XMLHTTP.6.0")))
    {
        m_szStatus = _T("GetXML().request.CreateInstance() is failed.");
        return bstrText;
    }
    hr = request->open(OLESTR("GET"), szUrl.AllocSysString(), _variant_t(VARIANT_FALSE), _variant_t(""), _variant_t(""));
    if (FAILED(hr))
    {
        m_szStatus = _T("GetXML().request.open() is failed.");
        szContent.Format(_T("%ld"),GetLastError());
        return bstrText;
    }
    if (SUCCEEDED(request->send(_variant_t(""))))
        request->get_responseText(&bstrText);
    else
        m_szStatus = _T("GetXML().request.Send() is failed.");        

    return bstrText;
}
