自绘按钮的两种形式

1. 窗口直接自绘

//按钮区域检测
void CMFCDlg::OnMouseMove(UINT nFlags, CPoint point)
{
    CRect rect;
    CWnd* pWnd = GetDlgItem(IDC_BTN_BMP);
    pWnd->GetWindowRect(rect);
    this->ScreenToClient(rect);
    CString str;
    str.Format(_T("x = %d, y = %d, rect: %d, %d"), point.x, point.y, rect.left, rect.top);
    SetWindowText(str);

    CDialogEx::OnMouseMove(nFlags, point);
}

void CMFCDlg::OnDrawItem(int nIDCtl, LPDRAWITEMSTRUCT lpDIS)
{
    switch (nIDCtl)
    {
    case IDOK:
        SetIconBtn(IDOK, IDI_ICON_OK, lpDIS);
        break;
    case IDCANCEL:
        SetIconBtn(IDCANCEL, IDI_ICON_CANCEL,lpDIS);
        break;
    case IDC_BTN_BMP:
        UINT nIDbmp[3] = { IDB_BMP1,IDB_BMP2,IDB_BMP3 };
        SetBmpBtn(IDC_BTN_BMP, nIDbmp, lpDIS);
        break;
    }
    CDialogEx::OnDrawItem(nIDCtl, lpDIS);
}

void CMFCDlg::SetIconBtn(UINT nIDCtl, UINT nIDIcon, LPDRAWITEMSTRUCT lpDIS)
{
    CDC dc;
    dc.Attach(lpDIS->hDC);
    HICON hIcon = theApp.LoadIcon(nIDIcon);
    CRect rect;
    GetDlgItem(nIDCtl)->GetClientRect(rect);
    if (!mdc.GetSafeHdc())
        mdc.Create(rect.Width(), rect.Height());
    mdc.SelectStockObject(WHITE_BRUSH);
    mdc.Rectangle(rect);
    if (lpDIS->itemState & ODS_SELECTED)
        rect.OffsetRect(1, 1);
    int x = rect.left + (rect.Height() - 32) / 2, y = x;
    mdc.DrawIcon(x, y, hIcon);
    CString str;
    GetDlgItemText(nIDCtl, str);
    rect.left += 40 + x * 2;
    mdc.DrawText(str, rect, DT_VCENTER | DT_SINGLELINE);
    dc.BitBlt(0, 0, lpDIS->rcItem.right, lpDIS->rcItem.bottom, &mdc, 0, 0, SRCCOPY);
    dc.Detach();
}

void CMFCDlg::SetBmpBtn(UINT nIDCtl, UINT nIDBmp[3], LPDRAWITEMSTRUCT lpDIS)
{
    CDC* pDC = CDC::FromHandle(lpDIS->hDC);
    CWnd* pWnd = GetDlgItem(nIDCtl);
    CRect rect(lpDIS->rcItem);

    if(!m_dcNormal.GetSafeHdc())
    m_dcNormal.LoadBitmap(nIDBmp[0]);
    if(!m_dcSelect.GetSafeHdc())
    m_dcSelect.LoadBitmap(nIDBmp[1]);
    if(!m_dcTrack.GetSafeHdc())
    m_dcTrack.LoadBitmap(nIDBmp[2]);

    if (rect.Width() != m_dcNormal.Width() || rect.Height() != m_dcNormal.Height())
    {
        pWnd->SetWindowPos(this, 0, 0, m_dcNormal.Width(), m_dcNormal.Height(), SWP_NOMOVE|SWP_NOZORDER);
        pWnd->GetClientRect(rect);
    }
    if (lpDIS->itemState & ODS_SELECTED)
    {
        pDC->BitBlt(0, 0, m_dcSelect.Width(), m_dcSelect.Height(), &m_dcSelect, 0, 0, SRCCOPY);
    }
    else
    {
        if(m_bBtnBmp)
            pDC->BitBlt(0, 0, m_dcTrack.Width(), m_dcTrack.Height(), &m_dcTrack, 0, 0, SRCCOPY);
        else
            pDC->BitBlt(0, 0, m_dcNormal.Width(), m_dcNormal.Height(), &m_dcNormal, 0, 0, SRCCOPY);
    }
    //因为当鼠标移动到控件上时，无法实时记录坐标点，所以没有找到解决办法2020年2月8日16点19分
}

BOOL CMFCDlg::OnSetCursor(CWnd* pWnd, UINT nHitTest, UINT message)
{
    HCURSOR hCursor = theApp.LoadStandardCursor(IDC_HAND);
    HCURSOR hCursor2 = theApp.LoadStandardCursor(IDC_ARROW);
    CWnd* pBtn = GetDlgItem(IDC_BTN_BMP);
    BOOL bTrack = FALSE;
    if (pBtn == pWnd)
    {
        SetCursor(hCursor);    
        bTrack = TRUE;
    }
    else
    {
        SetCursor(hCursor2);
        bTrack = FALSE;
    }
    if (m_bBtnBmp != bTrack)
    {
        m_bBtnBmp = bTrack;
        Invalidate(FALSE);
    }

    return TRUE;
    return CDialogEx::OnSetCursor(pWnd, nHitTest, message);
}

#include "pch.h"

// CIconBtnLx

IMPLEMENT_DYNAMIC(CBtnBmpLx, CButton)

CBtnBmpLx::CBtnBmpLx()
{
    m_bTrack = FALSE;
}

CBtnBmpLx::~CBtnBmpLx()
{
}

void CBtnBmpLx::DrawItem(LPDRAWITEMSTRUCT lpDIS)
{
    CDC* pDC = CDC::FromHandle(lpDIS->hDC);
    CRect rect(lpDIS->rcItem);
    CMemoryDC mdc(rect.Width(),rect.Height(),pDC);
    mdc.FillSolidRect(rect, m_clBack);
    //pDC->BitBlt(0, 0, mdc.Width(), mdc.Height(), &mdc, 0, 0, SRCCOPY);
    if (rect.Width() != m_dcNormal.Width() || rect.Height() != m_dcNormal.Height())
    {
        SetWindowPos(NULL, 0, 0, m_dcNormal.Width(), m_dcNormal.Height(), SWP_NOMOVE | SWP_NOZORDER);
        GetClientRect(rect);
    }

    if (lpDIS->itemState & ODS_SELECTED)
    {
        m_dcSelect.BitTrans(0, 0, mdc.Width(), mdc.Height(), &mdc, 0, 0, RGB(255, 255, 255));
        pDC->BitBlt(0, 0, m_dcSelect.Width(), m_dcSelect.Height(), &mdc, 0, 0, SRCCOPY);
    }
    else
    {
        if (m_dcTrack)
        {
            //CPoint point;
            //GetCursorPos(&point);
            //ScreenToClient(&point);
            if (m_bTrack)
            {
                m_dcTrack.BitTrans(0, 0, mdc.Width(), mdc.Height(), &mdc, 0, 0, RGB(255, 255, 255));
                pDC->BitBlt(0, 0, m_dcTrack.Width(), m_dcTrack.Height(), &mdc, 0, 0, SRCCOPY);
            }
            else
            {
                m_dcNormal.BitTrans(0, 0, mdc.Width(), mdc.Height(), &mdc, 0, 0, RGB(255, 255, 255));
                pDC->BitBlt(0, 0, m_dcNormal.Width(), m_dcNormal.Height(), &mdc, 0, 0, SRCCOPY);
            }
            return;
        }
        m_dcNormal.BitTrans(0, 0, mdc.Width(), mdc.Height(), &mdc, 0, 0, RGB(255, 255, 255));
        pDC->BitBlt(0, 0, m_dcNormal.Width(), m_dcNormal.Height(), &mdc, 0, 0, SRCCOPY);
    }

}


BEGIN_MESSAGE_MAP(CBtnBmpLx, CButton)
    ON_WM_MOUSEMOVE()

// CIconBtnLx 消息处理程序

int CBtnBmpLx::OnCreate(LPCREATESTRUCT lpCreateStruct)
{
    if (CButton::OnCreate(lpCreateStruct) == -1)
        return -1;

    return 0;
}


void CBtnBmpLx::SetBmpBtn(int nNormal, int nSelect, int nTrack)
{
    m_dcNormal.LoadBitmap(nNormal);
    m_dcSelect.LoadBitmap(nSelect);
    if (-1 == nTrack)
        return;
    m_dcTrack.LoadBitmap(nTrack);
}


void CBtnBmpLx::OnMouseMove(UINT nFlags, CPoint point)
{
    CRect rect;
    GetClientRect(rect);
    CDC* pDC = GetWindowDC();
    BOOL bTrack = FALSE;
    if(rect.PtInRect(point))
    {
        bTrack = TRUE;
        pDC->BitBlt(0, 0, m_dcTrack.Width(), m_dcTrack.Height(), &m_dcTrack, 0, 0, SRCCOPY);
        if (GetCapture() != this)
            SetCapture();
    }
    else
    {
        ReleaseCapture();
    }
    if (bTrack != m_bTrack)
    {
        m_bTrack = bTrack;
        Invalidate(FALSE);
    }
}

void CBtnIconLx::DrawItem(LPDRAWITEMSTRUCT lpDIS)
{
    CDC dc;
    dc.Attach(lpDIS->hDC);
    //CDC* pDC = CDC::FromHandle(lpDIS->hDC);
    CRect rect(lpDIS->rcItem);
    if(!m_dc.GetSafeHdc())
        m_dc.Create(rect.Width(),rect.Height(),&dc);
    m_dc.SelectStockObject(WHITE_BRUSH);
    m_dc.Rectangle(&lpDIS->rcItem);
    if(lpDIS->itemState & ODS_SELECTED)
        rect.OffsetRect(1,1);
    m_dc.SelectObject(dc.GetCurrentFont());
    int x = rect.left + (rect.Height() - 32)/2, y = x;
    m_dc.DrawIcon(x, y, m_hIcon);
    CString str;
    this->GetWindowText(str);
    rect.left += 32 + x * 2;
    m_dc.DrawText(str, rect, DT_VCENTER | DT_SINGLELINE);
    dc.BitBlt(0,0,lpDIS->rcItem.right,lpDIS->rcItem.bottom,&m_dc,0,0,SRCCOPY);
    dc.Detach();
}
