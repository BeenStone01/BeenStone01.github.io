void DlgFileOpen::OnBnClickedBtnOpen()
{
    BOOL isOpen = TRUE;
    CString defaultDir = L"E:\\FileText";
    CString  fileName = L"";
    CString fileType = L"文件（*.doc; *.ppt; *.xls)|*.odc;*.ppt;*.xls||";
    CFileDialog openFileDlg(isOpen, defaultDir, fileName, OFN_HIDEREADONLY | OFN_READONLY, fileType, NULL);
    openFileDlg.GetOFN().lpstrInitialDir = L"E:\\FileText\test.doc";
    INT_PTR result = openFileDlg.DoModal();
    CString filePath = defaultDir + "\\test.doc";
    if (result == IDOK)
    {
        filePath = openFileDlg.GetPathName();
    }
    CWnd::SetDlgItemTextW(IDC_EDIT_OPEN, filePath);
}

void DlgFileOpen::OnBnClickedBtnSave()
{
    BOOL isOpen = FALSE;
    CString defaultDir = L"E:\\FileTest";
    CString fileName = L"test.doc";
    CString fileType = L"文件（*.doc; *.ppt; *.xls)|*.doc;*.ppt;*.xls||";
    CFileDialog openFileDlg(isOpen, defaultDir, fileName, OFN_HIDEREADONLY | OFN_OVERWRITEPROMPT, fileType, NULL);
    openFileDlg.GetOFN().lpstrInitialDir = L"E:\\FileTest\\test.doc";
    INT_PTR result = openFileDlg.DoModal();
    CString filePath = defaultDir + "\\" + fileName;
    if (result == IDOK)
    {
        filePath = openFileDlg.GetPathName();
    }
    CWnd::SetDlgItemTextW(IDC_EDIT_SAVE, filePath);
}
